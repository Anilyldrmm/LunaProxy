> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Walk kütüphanesini tamamen kaldır; Win32 WS_POPUP + go-webview2 + HTML/CSS/JS ile profesyonel dark-theme UI, per-IP cihaz takibi, QR→/setup yönlendirme ve GitHub oto-güncelleme sistemi kur.

**Tech Stack:** Go 1.21, `github.com/jchv/go-webview2` (CGO/mingw-w64), `github.com/getlantern/systray`, Win32 user32/dwmapi, HTML/CSS/JS (tek dosya, go:embed)

**Build:** `CGO_ENABLED=1 CC=gcc go build -tags withbundle -ldflags "-X main.Version=0.1.0 -H windowsgui" -o SpAC3DPI.exe .`

```go
type ghRelease struct {
    TagName string `json:"tag_name"`
    Assets  []struct {
        Name               string `json:"name"`
        BrowserDownloadURL string `json:"browser_download_url"`
    } `json:"assets"`
}

// newerVersion — latest, current'tan büyükse true döner.
func newerVersion(current, latest string) bool {
    cur := parseVer(strings.TrimPrefix(current, "v"))
    lat := parseVer(strings.TrimPrefix(latest, "v"))
    for i := 0; i < 3; i++ {
        if lat[i] > cur[i] {
            return true
        }
        if lat[i] < cur[i] {
            return false
        }
    }
    return false
}

func parseVer(s string) [3]int {
    parts := strings.SplitN(s, ".", 3)
    var v [3]int
    for i, p := range parts {
        if i >= 3 {
            break
        }
        n, _ := strconv.Atoi(p)
        v[i] = n
    }
    return v
}

// CheckUpdate — GitHub'dan son sürümü kontrol eder.
// Güncelleme varsa (tagName, downloadURL) döner; yoksa ("","") döner.
func CheckUpdate() (tagName, downloadURL string, err error) {
    url := fmt.Sprintf("https://api.github.com/repos/%s/%s/releases/latest",
        githubOwner, githubRepo)
    client := &http.Client{Timeout: 10 * time.Second}
    resp, err := client.Get(url)
    if err != nil {
        return "", "", err
    }
    defer resp.Body.Close()
    if resp.StatusCode != 200 {
        return "", "", fmt.Errorf("GitHub API: %d", resp.StatusCode)
    }
    var rel ghRelease
    if err := json.NewDecoder(resp.Body).Decode(&rel); err != nil {
        return "", "", err
    }
    if !newerVersion(Version, rel.TagName) {
        return "", "", nil
    }
    for _, a := range rel.Assets {
        if strings.EqualFold(a.Name, "SpAC3DPI.exe") {
            return rel.TagName, a.BrowserDownloadURL, nil
        }
    }
    return rel.TagName, "", nil
}

// DownloadAndReplace — yeni exe'yi indirir, PS1 replace script çalıştırır, çıkar.
func DownloadAndReplace(downloadURL string) error {
    tmpDir := os.TempDir()
    newExe := filepath.Join(tmpDir, "SpAC3DPI_update.exe")

    // İndir
    resp, err := http.Get(downloadURL)
    if err != nil {
        return fmt.Errorf("indirme hatası: %w", err)
    }
    defer resp.Body.Close()
    f, err := os.Create(newExe)
    if err != nil {
        return err
    }
    if _, err := io.Copy(f, resp.Body); err != nil {
        f.Close()
        return err
    }
    f.Close()

    // Mevcut exe yolu
    selfExe, err := os.Executable()
    if err != nil {
        return err
    }

    // PowerShell replace script
    ps1 := filepath.Join(tmpDir, "SpAC3DPI_update.ps1")
    script := fmt.Sprintf(`Start-Sleep -Seconds 2
Copy-Item -Force "%s" "%s"
Start-Process "%s"
Remove-Item "%s" -ErrorAction SilentlyContinue
`, newExe, selfExe, selfExe, ps1)
    if err := os.WriteFile(ps1, []byte(script), 0644); err != nil {
        return err
    }

    // Script'i gizli çalıştır
    cmd := exec.Command("powershell.exe", "-WindowStyle", "Hidden", "-ExecutionPolicy", "Bypass", "-File", ps1)
    cmd.SysProcAttr = &syscall.SysProcAttr{HideWindow: true, CreationFlags: 0x08000000}
    if err := cmd.Start(); err != nil {
        return fmt.Errorf("güncelleme script başlatılamadı: %w", err)
    }
    os.Exit(0)
    return nil
}

// StartUpdateChecker — arka planda her 6 saatte bir güncelleme kontrol eder.
// Güncelleme bulunursa onNotify(tagName) çağrılır.
func StartUpdateChecker(onNotify func(tagName string)) {
    go func() {
        check := func() {
            tag, _, err := CheckUpdate()
            if err != nil {
                logWarn("Güncelleme kontrolü başarısız: " + err.Error())
                return
            }
            if tag != "" {
                logInfo("Yeni sürüm mevcut: " + tag)
                onNotify(tag)
            }
        }
        check() // başlangıçta hemen kontrol
        ticker := time.NewTicker(6 * time.Hour)
        defer ticker.Stop()
        for range ticker.C {
            check()
        }
    }()
}
```

```go
// ── Per-IP cihaz takibi ──────────────────────────────────────────────────────

type deviceEntry struct {
    Bytes       int64
    ActiveConns int64
}

var devices sync.Map // key: string IP, value: *deviceEntry

// DeviceInfo — UI'a gönderilen cihaz bilgisi.
type DeviceInfo struct {
    IP          string `json:"ip"`
    Bytes       int64  `json:"bytes"`
    ActiveConns int64  `json:"activeConns"`
}

func trackDevice(ip string, bytes int64) {
    v, _ := devices.LoadOrStore(ip, &deviceEntry{})
    e := v.(*deviceEntry)
    atomic.AddInt64(&e.Bytes, bytes)
}

func incDeviceConn(ip string) {
    v, _ := devices.LoadOrStore(ip, &deviceEntry{})
    atomic.AddInt64(&v.(*deviceEntry).ActiveConns, 1)
}

func decDeviceConn(ip string) {
    if v, ok := devices.Load(ip); ok {
        atomic.AddInt64(&v.(*deviceEntry).ActiveConns, -1)
    }
}

func GetDevices() []DeviceInfo {
    var list []DeviceInfo
    devices.Range(func(k, v any) bool {
        e := v.(*deviceEntry)
        list = append(list, DeviceInfo{
            IP:          k.(string),
            Bytes:       atomic.LoadInt64(&e.Bytes),
            ActiveConns: atomic.LoadInt64(&e.ActiveConns),
        })
        return true
    })
    return list
}

func remoteIP(addr string) string {
    if i := strings.LastIndex(addr, ":"); i >= 0 {
        return addr[:i]
    }
    return addr
}
```

```go
    }
    defer wv.Destroy()

    wv.SetTitle("SpAC3DPI")
    wv.SetSize(400, 640, webview.HintFixed)

    hwnd := uintptr(wv.Window())
    makeFrameless(hwnd)
    centerWindow(hwnd, 400, 640)
    applyDWMShadow(hwnd)

    // UI HTML yükle
    html := string(uiHTMLBytes)
    wv.SetHtml(html)

    // IPC mesaj handler'ı bağla
    wv.Bind("goMessage", handleIPCMessage)

    // Status/log push ticker başlat
    go startIPCTicker()

    wv.Run()
}

func makeFrameless(hwnd uintptr) {
    style, _, _ := procGetWindowLong.Call(hwnd, uintptr(uint32(gwlStyle)))
    // WS_POPUP yap, caption/sysmenu/thickframe/minbox/maxbox kaldır
    newStyle := (uint32(style) &^ (wsCaption | wsSysMenu | wsThickFrame | wsMinBox | wsMaxBox)) | wsPopup
    procSetWindowLong.Call(hwnd, uintptr(uint32(gwlStyle)), uintptr(newStyle))
    procSetWindowPos.Call(hwnd, 0, 0, 0, 0, 0,
        uintptr(swpNoMove|swpNoSize|swpNoZOrder|swpFrameChg))
}

func centerWindow(hwnd uintptr, w, h int) {
    sw, _, _ := procGetSystemMetrics.Call(smCxScreen)
    sh, _, _ := procGetSystemMetrics.Call(smCyScreen)
    x := (int(sw) - w) / 2
    y := (int(sh) - h) / 2
    procSetWindowPos.Call(hwnd, 0, uintptr(x), uintptr(y), uintptr(w), uintptr(h),
        uintptr(swpNoZOrder|swpFrameChg))
}

func applyDWMShadow(hwnd uintptr) {
    // Windows 11: border rengi kaldır → DWM kendi shadow'unu ekler
    color := uint32(dwmwaColorNone)
    procDwmSetWindowAttr.Call(hwnd, uintptr(dwmwaBorderColor),
        uintptr(unsafe.Pointer(&color)), 4)
}

// showWindow — pencereyi göster (tray'den çağrılır).
func showWindow() {
    if wv == nil {
        return
    }
    hwnd := uintptr(wv.Window())
    procShowWindow.Call(hwnd, swShow)
}

// hideWindow — pencereyi gizle (tray'e minimize).
func hideWindow() {
    if wv == nil {
        return
    }
    hwnd := uintptr(wv.Window())
    procShowWindow.Call(hwnd, swHide)
}

// evalJS — WebView thread'inde güvenli JS eval.
func evalJS(js string) {
    if wv == nil {
        return
    }
    wv.Dispatch(func() {
        wv.Eval(js)
    })
}

// jsonEscape — basit JSON string escape (tırnak ve ters eğik çizgi için).
func jsonEscape(s string) string {
    s = strings.ReplaceAll(s, `\`, `\\`)
    s = strings.ReplaceAll(s, `"`, `\"`)
    s = strings.ReplaceAll(s, "\n", `\n`)
    s = strings.ReplaceAll(s, "\r", ``)
    return fmt.Sprintf(`"%s"`, s)
}
```

```go
// handleIPCMessage — JS'den gelen postMessage'ları işler.
// wv.Bind("goMessage", handleIPCMessage) ile kayıtlı.
func handleIPCMessage(data string) {
    var msg struct {
        Type    string          `json:"type"`
        Payload json.RawMessage `json:"payload"`
    }
    if err := json.Unmarshal([]byte(data), &msg); err != nil {
        logWarn("IPC parse hatası: " + err.Error())
        return
    }
    switch msg.Type {
    case "toggle":
        if g.running {
            g.stop()
        } else {
            if err := g.start(); err != nil {
                logError("Başlatma hatası: " + err.Error())
                evalJS(fmt.Sprintf(`showError(%s)`, jsonEscape(err.Error())))
            }
        }
        pushStatus()

    case "saveSettings":
        var cfg Config
        if err := json.Unmarshal(msg.Payload, &cfg); err != nil {
            logWarn("saveSettings parse: " + err.Error())
            return
        }
        setConfig(cfg)
        setStartup(cfg.AutoStart)
        evalJS(`showSaveSuccess()`)
        if g.running {
            g.restart()
        }

    case "clearLogs":
        logBuffer.Clear()

    case "copyToClipboard":
        var p struct{ Text string `json:"text"` }
        json.Unmarshal(msg.Payload, &p)
        setClipboard(p.Text)

    case "windowMinimize":
        hwnd := uintptr(wv.Window())
        procShowWindow.Call(hwnd, 2) // SW_MINIMIZE

    case "windowHide":
        hideWindow()

    case "windowExit":
        appExiting = true
        g.shutdown()
        os.Exit(0)

    case "startDrag":
        hwnd := uintptr(wv.Window())
        procReleaseCapture.Call()
        procSendMessage.Call(hwnd, 0x00A1, 2, 0) // WM_NCLBUTTONDOWN, HTCAPTION

    case "requestQR":
        pushQR()

    case "requestSettings":
        pushSettings()

    case "openFileDialog":
        // Dosya seçici — ileride implemente edilecek
        evalJS(`fileSelected("gdpiPath", "")`)

    case "applyUpdate":
        var p struct{ URL string `json:"url"` }
        json.Unmarshal(msg.Payload, &p)
        if err := DownloadAndReplace(p.URL); err != nil {
            logError("Güncelleme başarısız: " + err.Error())
        }
    }
}
```

```go
                g.shutdown()
                systray.Quit()

        trayStart.Hide()

    } else {
        trayStop.Hide()
        trayStart.Show()
    }
}
```

```go
    initTray()

    // Oto-güncelleme kontrolü
    StartUpdateChecker(func(tag string) {
        // pendingUpdateTag'i set et; bir sonraki pushStatus'ta UI'a gider
        pendingUpdateTag.Store(tag)
    })

    // Logo bytes inject — WebView2 init'ten önce window.__logoB64 hazır olsun
    logoB64 := logoBase64()
    _ = logoB64 // initWindow içinde kullanılacak

    initWindow() // bloklar (webview.Run())
}
```

`webview_win.go` içindeki `initWindow()`'a logo inject ekle:

```go
    // Logo inject — wv.SetHtml'den önce
    logoB64 := logoBase64()
    wv.Init(fmt.Sprintf(`window.__logoB64 = "%s";`, logoB64))
```

```go
import "encoding/base64"

func logoBase64() string {

}
```
